<script>
    function openDoor() {
        document.getElementById('door-overlay').classList.add('door-open');
        setTimeout(() => document.getElementById('door-overlay').style.display = 'none', 800);
    }

    function addItem(btn) {
        const row = document.createElement('div');
        row.className = 'item-row';
        row.innerHTML = `
            <input type="number" placeholder="0.00" class="p-item">
            <button class="btn-danger" style="font-size:14px" onclick="this.parentElement.remove()">×</button>
        `;
        btn.previousElementSibling.appendChild(row);
    }

    function addPerson() {
        const container = document.getElementById('people-container');
        const count = document.querySelectorAll('.person-block').length + 1;
        const div = document.createElement('div');
        div.className = 'person-block';
        div.innerHTML = `
            <div class="person-header">
                <input type="text" class="p-name" value="Member ${count}">
                <button class="btn-danger" onclick="this.parentElement.parentElement.remove()">×</button>
            </div>
            <div class="items-list">
                <div class="item-row"><input type="number" placeholder="0.00" class="p-item"></div>
            </div>
            <button class="btn btn-add-item" onclick="addItem(this)">+ Add Item</button>
        `;
        container.appendChild(div);
    }

    function calculate() {
        const blocks = document.querySelectorAll('.person-block');
        let participants = [];
        let grandTotal = 0;

        blocks.forEach(block => {
            let name = block.querySelector('.p-name').value || "Unnamed";
            let items = block.querySelectorAll('.p-item');
            let totalPaid = 0;
            items.forEach(input => totalPaid += (parseFloat(input.value) || 0));
            
            grandTotal += totalPaid;
            participants.push({ name, paid: totalPaid });
        });

        if (grandTotal === 0) return alert("Enter at least one amount!");

        const share = grandTotal / participants.length;
        document.getElementById('res-total').innerText = "₹" + grandTotal.toFixed(2);
        document.getElementById('res-share').innerText = "₹" + share.toFixed(2);

        let balances = participants.map(p => ({ name: p.name, amount: p.paid - share }));
        let debtors = balances.filter(b => b.amount < -0.01).sort((a,b) => a.amount - b.amount);
        let creditors = balances.filter(b => b.amount > 0.01).sort((a,b) => b.amount - a.amount);

        const list = document.getElementById('settlement-list');
        list.innerHTML = "<h4 style='margin:0 0 10px 0; font-size:14px'>Settlement Plan:</h4>";

        let d=0, c=0;
        while(d < debtors.length && c < creditors.length) {
            let pay = Math.min(Math.abs(debtors[d].amount), creditors[c].amount);
            list.innerHTML += `<div class="settlement-item"><b>${debtors[d].name}</b> pays <b>₹${pay.toFixed(2)}</b> to <b>${creditors[c].name}</b></div>`;
            debtors[d].amount += pay;
            creditors[c].amount -= pay;
            if(Math.abs(debtors[d].amount) < 0.01) d++;
            if(creditors[c].amount < 0.01) c++;
        }

        if(debtors.length === 0 && creditors.length === 0) {
            list.innerHTML += "<p style='text-align:center; color:#22c55e'>Everyone is settled! 🎉</p>";
        }
        
        document.getElementById('results').style.display = 'block';
    }
</script>
